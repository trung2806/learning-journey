### `srl_engine.py`

```python
from dataclasses import dataclass
from pathlib import Path
from typing import Any
import yaml


@dataclass(frozen=True)
class CpuAggregate:
    """Aggregate (index='all') CPU utilization from SR Linux platform state.

    instant and average_1 are integer percent (0-100), already computed by the
    device. instant is a point sample; average_1 is the 1-minute average.
    """

    instant: int
    average_1: int


def parse_cpu_aggregate(raw_json: dict[str, Any]) -> CpuAggregate:
    """Extract aggregate CPU% from `info from state platform control * | as json`.

    Pure logic, no I/O. Every missing or ill-shaped field raises with a precise
    message instead of defaulting, so a schema drift fails loud and never reads
    as a silent 0% (a swallowed alert).
    """
    if not isinstance(raw_json, dict):
        raise ValueError(f"expected dict payload, got {type(raw_json).__name__}")

    controls = raw_json.get("control")
    if not isinstance(controls, list) or not controls:
        raise ValueError("payload has no non-empty 'control' list")

    control = controls[0]
    if not isinstance(control, dict):
        raise ValueError("control[0] is not an object")

    cpu_list = control.get("cpu")
    if not isinstance(cpu_list, list) or not cpu_list:
        raise ValueError("control[0] has no non-empty 'cpu' list")

    aggregate = next(
        (c for c in cpu_list if isinstance(c, dict) and c.get("index") == "all"),
        None,
    )
    if aggregate is None:
        raise ValueError("no cpu entry with index 'all'")

    total = aggregate.get("total")
    if not isinstance(total, dict):
        raise ValueError("aggregate cpu entry has no 'total' object")

    instant = total.get("instant")
    average_1 = total.get("average-1")
    if instant is None or average_1 is None:
        raise ValueError("aggregate 'total' is missing 'instant' or 'average-1'")

    try:
        instant, average_1 = int(instant), int(average_1)
    except (ValueError, TypeError):
        raise ValueError(
            f"CPU values must be integers, got instant={type(instant).__name__}, average-1={type(average_1).__name__}"
        )

    for name, value in (("instant", instant), ("average-1", average_1)):
        if not 0 <= value <= 100:
            raise ValueError(f"'{name}' out of percent range 0-100: {value}")

    return CpuAggregate(instant=instant, average_1=average_1)


def parse_memory_utilization(raw_json: dict[str, Any]) -> int:
    """Extract platform control memory utilization percentage.

    Applies strict fail-loud constraints. Does not use a one-field dataclass.
    Returns raw scalar integer (0-100).
    """
    if not isinstance(raw_json, dict):
        raise ValueError(f"expected dict payload, got {type(raw_json).__name__}")

    controls = raw_json.get("control")
    if not isinstance(controls, list) or not controls:
        raise ValueError("payload has no non-empty 'control' list")

    control = controls[0]
    if not isinstance(control, dict):
        raise ValueError("control[0] is not an object")

    memory = control.get("memory")
    if not isinstance(memory, dict):
        raise ValueError("control[0] has no 'memory' object")

    utilization = memory.get("utilization")
    if utilization is None:
        raise ValueError("memory object is missing 'utilization'")

    try:
        util_int = int(utilization)
    except (ValueError, TypeError):
        raise ValueError(
            f"memory 'utilization' must be an integer, got {type(utilization).__name__}"
        )

    if not 0 <= util_int <= 100:
        raise ValueError(f"'utilization' out of percent range 0-100: {util_int}")

    return util_int


def validate_inventory_credentials(inventory: list[dict]) -> None:
    """Kiểm tra toàn vẹn khâu render tài khoản của inventory.

    Chỉ quét token '{{' vì đây là ký tự đại diện cho biến chứa secret chưa
    giải bọc.
    """
    leaks = [d.get("host") for d in inventory if "{{" in repr(d.get("password"))]
    if leaks:
        raise ValueError(f"Unrendered credential template leaked on hosts: {leaks}")


def load_inventory(path: Path) -> list[dict]:
    """Load inventory từ file YAML và kích hoạt kiểm tra rò rỉ credential template.

    Áp dụng nguyên lý Structured-at-Source và bảo vệ nghiêm ngặt khế ước cấu
    trúc (shape validation) để tránh AttributeError khi file rỗng hoặc sai
    định dạng mapping.
    """
    with open(path, "r", encoding="utf-8") as f:
        inventory = yaml.safe_load(f)

    if not isinstance(inventory, list):
        raise ValueError(
            f"inventory at {path} must be a top-level list of host mappings, "
            f"got {type(inventory).__name__}"
        )

    validate_inventory_credentials(inventory)
    return inventory
```
