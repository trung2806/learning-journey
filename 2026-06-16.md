- **Pre-computed percent (%-ăn-liền)**: metric thiết bị tự tính sẵn ra % nguyên (CPU `total`, `memory.utilization`), extract thẳng, không derive. Đối lập cumulative counter (phải sample hai lần qua một interval).

### `srl_engine.py`

```python
from dataclasses import dataclass
from typing import Any
from pathlib import Path
import json


@dataclass(frozen=True)
class CpuAggregate:
    """Aggregate (index='all') CPU utilization từ SR Linux platform state.

    instant và average_1 là integer percent (0-100), device đã tự tính.
    instant là point sample; average_1 là 1-minute average.
    """
    instant: int
    average_1: int


def parse_cpu_aggregate(raw_json: dict[str, Any]) -> CpuAggregate:
    """Extract aggregate CPU% từ `info from state platform control * | as json`.

    Pure logic, no I/O. Mọi field thiếu/sai shape RAISE với message chính xác thay
    vì default, để schema drift fail loud, không bao giờ đọc thành 0% (swallowed alert).
    """
    if not isinstance(raw_json, dict):
        raise ValueError(f"expected dict payload, got {type(raw_json).__name__}")

    controls = raw_json.get("control")
    if not isinstance(controls, list) or not controls:
        raise ValueError("payload has no non-empty 'control' list")

    control = controls[0]
    if not isinstance(control, dict):
        raise ValueError("control[0] is not an object")

    cpu_list = control.get("cpu")
    if not isinstance(cpu_list, list) or not cpu_list:
        raise ValueError("control[0] has no non-empty 'cpu' list")

    aggregate = next((c for c in cpu_list if isinstance(c, dict) and c.get("index") == "all"), None)
    if aggregate is None:
        raise ValueError("no cpu entry with index 'all'")

    total = aggregate.get("total")
    if not isinstance(total, dict):
        raise ValueError("aggregate cpu entry has no 'total' object")

    instant = total.get("instant")
    average_1 = total.get("average-1")
    if instant is None or average_1 is None:
        raise ValueError("aggregate 'total' is missing 'instant' or 'average-1'")

    instant, average_1 = int(instant), int(average_1)
    for name, value in (("instant", instant), ("average-1", average_1)):
        if not 0 <= value <= 100:
            raise ValueError(f"'{name}' out of percent range 0-100: {value}")

    return CpuAggregate(instant=instant, average_1=average_1)


def validate_inventory_credentials(inventory: list[dict]) -> None:
    """Kiểm tra toàn vẹn khâu render tài khoản của inventory.

    Chỉ quét token '{{' vì đây là biến substitution chứa secret chưa giải bọc
    ('{%' là control flow, không ôm credential). Không dùng 'assert' (bị triệt
    tiêu dưới `python -O` → guard biến mất); raise vô điều kiện, gom hết host.
    """
    leaks = [d.get("host") for d in inventory if "{{" in repr(d.get("password"))]
    if leaks:
        raise ValueError(f"Unrendered credential template leaked on hosts: {leaks}")


def load_inventory(path: Path) -> list[dict]:
    """Load inventory rồi kích hoạt guard credential ngay trên load path.

    LƯU Ý (open item Day 20): json.load giả định inventory là JSON. Nếu inventory
    thật là YAML, đổi sang yaml.safe_load(f), nếu không sẽ nổ trên file YAML.
    """
    with open(path, "r") as f:
        inventory = json.load(f)
    validate_inventory_credentials(inventory)
    return inventory
```

def test_missing_total_raises_not_zero():
    """Detonation: regression .get(k,0) sẽ trả 0 (xanh sai); bản fail-loud đỏ."""
    payload_drift = {"control": [{"cpu": [{"index": "all"}]}]}
    with pytest.raises(ValueError) as exc_info:
        parse_cpu_aggregate(payload_drift)
    assert "has no 'total' object" in str(exc_info.value)

def test_selects_all_not_first_core():
    """Detonation: parser dùng cpu_list[0] sẽ chọn nhầm core; phải quét index='all'."""

        {"index": "all", "total": {"instant": 7,  "average-1": 5}},
    ]}]}
    assert parse_cpu_aggregate(payload_multi_core) == CpuAggregate(instant=7, average_1=5)
