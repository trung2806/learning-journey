```python
    )

        print(f"Error: {args.input} not found", file=sys.stderr)
        return 1

    vlans = load_vlans(args.input)
    config = render_config(vlans)

    if args.output is None:
        print(config)
    else:
        args.output.write_text(config, encoding="utf-8")
        print(f"Wrote {args.output}", file=sys.stderr)

    return 0
```

## find_stale.py — Bash refactor #4 (`find -mtime`)

### Bản tự viết đầu (engine, đúng logic nhưng thiếu)
```python
import time
from pathlib import Path

def find_old_files(directory, n_days):
    n_seconds = n_days * 86400
    current_time = time.time()
    dir_path = Path(directory)
    for file_path in dir_path.rglob('*'):
        if file_path.is_file():
            file_mtime = file_path.stat().st_mtime
            if (current_time - file_mtime) > n_seconds:
                print(file_path.absolute())
```
- Logic đúng: mtime và `time.time()` đều là epoch giây, hiệu > N*86400 = cũ hơn N ngày.
- Thiếu so với spec: (1) chưa có argparse → chưa thành CLI `find_stale.py <dir> --days N`, mà đây mới là rep argparse thứ 2 của ngày, mình suýt skip; (2) bỏ cột tuổi (spec yêu cầu in tuổi theo ngày).

### Bản đóng (tách engine/I-O, thêm argparse + cột tuổi)
```python
#!/usr/bin/env python3
"""Tim file cu hon N ngay trong 1 thu muc (Python thay cho `find -mtime`)."""
import argparse
import sys
import time
from pathlib import Path


def find_old_files(directory: Path, n_days: int) -> list[tuple[Path, int]]:
    """Tra ve list (duong_dan, tuoi_ngay) cua file cu hon n_days."""
    n_seconds = n_days * 86400
    now = time.time()
    result = []
    for path in directory.rglob("*"):
        if path.is_file():
            age_seconds = now - path.stat().st_mtime
            if age_seconds > n_seconds:
                result.append((path, int(age_seconds // 86400)))
    return result


def main() -> int:
    parser = argparse.ArgumentParser(
        description="Tim file cu hon N ngay (thay cho find -type f -mtime +N)."
    )
    parser.add_argument("directory", type=Path, help="Thu muc can quet")
    parser.add_argument(
        "--days", type=int, default=30, help="Nguong tuoi tinh bang ngay (default 30)"
    )
    args = parser.parse_args()

    if not args.directory.is_dir():
        print(f"Error: {args.directory} khong phai thu muc", file=sys.stderr)
        return 1

    for path, age_days in find_old_files(args.directory, args.days):
        print(f"{age_days:>5}d  {path}")

    return 0


if __name__ == "__main__":
    sys.exit(main())
```

- `--days type=int` bắt buộc, quên `type` thì so int với string sẽ crash.

## Glossary thêm (2 mục)
- **`Path.rglob("*")`**: duyệt đệ quy mọi mục trong cây thư mục con. Lọc file thật bằng `p.is_file()`. (Bản không đệ quy: `iterdir()` / `glob("*")`.)
- **Tuổi file từ mtime**: `p.stat().st_mtime` ra epoch giây; `time.time() - mtime` ra tuổi (giây). Đổi ngày: `// 86400`. (Cách object hơn: `datetime.fromtimestamp(mtime)`, trừ 2 datetime ra `timedelta`, lấy `.days`.)
